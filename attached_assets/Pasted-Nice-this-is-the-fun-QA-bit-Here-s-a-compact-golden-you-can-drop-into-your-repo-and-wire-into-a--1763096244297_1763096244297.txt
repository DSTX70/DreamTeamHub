### How you might use this in your Jest/Vitest harness

Pseudo-code (you don’t have to do this right now, just to show how it plugs in):

```ts
import fs from 'node:fs';
import path from 'node:path';
import { LifestyleBannerPackSchema } from '../../server/ai/schemas/lifestyleBannerPack';
import { EcomPdpAplusContentPackSchema } from '../../server/ai/schemas/ecomPdpAplusContentPack';

const golden = JSON.parse(
  fs.readFileSync(
    path.join(__dirname, 'goldens', 'wo002_colorcue.json'),
    'utf8'
  )
);

describe('WO-002 goldens', () => {
  test('Lifestyle Pack meets golden expectations', async () => {
    const { work_item, goldens } = golden;
    const { lifestyle_pack } = goldens;

    // 1. Seed a test WI with work_item.title/body
    // 2. Run generateLifestyleBannerPack for that WI
    const output = await runLifestylePackForWorkItem(work_item); // your helper

    const parsed = LifestyleBannerPackSchema.parse(output);

    // 3. Structural checks
    expect(parsed.entries.length).toBeGreaterThanOrEqual(lifestyle_pack.min_entries);
    expect(parsed.entries.length).toBeLessThanOrEqual(lifestyle_pack.max_entries);

    const allowed = new Set(lifestyle_pack.allowed_skus);
    for (const entry of parsed.entries) {
      expect(allowed.has(entry.sku)).toBe(true);
      lifestyle_pack.required_fields_per_entry.forEach((field) => {
        expect(entry[field]).toBeTruthy();
      });

      const expectedSlot = lifestyle_pack.expected_hero_slots[entry.sku];
      if (expectedSlot) {
        expect(entry.hero_slot).toBe(expectedSlot);
      }

      expect(entry.filename).toContain(entry.sku);
      expect(entry.filename).toContain(lifestyle_pack.filename_expectations.must_contain_resolution_hint);
    }
  });

  test('PDP Pack meets golden expectations', async () => {
    const { work_item, goldens } = golden;
    const { pdp_pack } = goldens;

    const output = await runPdpPackForWorkItem(work_item); // your helper
    const parsed = EcomPdpAplusContentPackSchema.parse(output);

    pdp_pack.required_top_level_fields.forEach((field) => {
      expect(parsed[field]).toBeTruthy();
    });

    expect(parsed.feature_bullets.length).toBeGreaterThanOrEqual(
      pdp_pack.feature_bullets_expectations.min_count
    );

    expect(parsed.faqs.length).toBeGreaterThanOrEqual(
      pdp_pack.faqs_expectations.min_count
    );

    // And so on: check slug, A+ sections, etc.
  });
});
```

The key is: the JSON above encodes:

* **The WO-002 brief** (so the test is self-contained), and
* **Formal expectations** for Lifestyle + PDP packs:

  * **Which SKUs are allowed**
  * **Minimum structure and counts**
  * **Field-level expectations** (hero slots, filename patterns, bullets, etc.)

So now, every time you change a prompt, schema, or product catalog wiring, you can run the golden test and immediately see if WO-002 still behaves the way you intended.
