## 1. Backend: add a simple “create work item for project” route

In your server routes (where other `/api` routes live), add something like:

```ts
// server/routes/projects.ts (or wherever your project routes are)
import { Router } from 'express';
import { db } from '../db';
import { workItems } from '../db/schema'; // Drizzle table
// ^ adapt import to your actual schema

const router = Router();

/**
 * Create a new Work Item attached to a project.
 * Body: { title: string; body?: string; lane?: string; priority?: string; dueDate?: string | null }
 */
router.post('/api/projects/:projectId/work-items', async (req, res, next) => {
  try {
    const projectId = req.params.projectId;
    const { title, body, lane, priority, dueDate } = req.body;

    if (!title) {
      return res.status(400).json({ error: 'title is required' });
    }

    const [wi] = await db
      .insert(workItems)
      .values({
        projectId,
        title,
        body: body ?? '',
        lane: lane ?? 'brand-assets',      // default lane; tweak to your enum
        priority: priority ?? 'normal',
        // parse dueDate if provided; avoid the “Expected date, received string” error
        dueDate: dueDate ? new Date(dueDate) : null,
        status: 'open',
      })
      .returning();

    return res.json(wi);
  } catch (err) {
    next(err);
  }
});

export default router;
```

And make sure this `projects` router is mounted in your main `server/routes.ts` if it isn’t already:

```ts
// server/routes.ts
import projectsRoutes from './routes/projects';

app.use(projectsRoutes);
```

> Key point: this gives us a clean `POST /api/projects/:projectId/work-items` endpoint the UI can call when you click “New Work Item.”

## 2. Frontend: add a “New Work Item” button to the Tasks tab

In your React component that renders the **Tasks** tab for a project (it looks like `ProjectTasks.tsx` or similar), you currently render the empty state:

> “No tasks created yet”

We’ll replace that with an empty state that includes a button.

```tsx
// app/components/projects/ProjectTasks.tsx (name approximate)
import React, { useState } from 'react';
import { Button } from '@/components/ui/button'; // or whatever button component you use

type ProjectTasksProps = {
  projectId: string;
  // tasks: WorkItem[];  // existing props
  onTaskCreated?: (task: any) => void;
};

export function ProjectTasksEmptyState({ projectId, onTaskCreated }: ProjectTasksProps) {
  const [creating, setCreating] = useState(false);

  async function handleCreateWorkItem() {
    try {
      setCreating(true);

      // Minimal starter WI – you’ll paste WO-001 body in the next step.
      const resp = await fetch(`/api/projects/${projectId}/work-items`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          title: 'Brand Guidelines – WO-001 Assets',
          body: '',         // we’ll paste the kickoff text manually in the UI after it opens
          lane: 'brand-assets',
        }),
      });

      if (!resp.ok) {
        const err = await resp.text();
        console.error('Failed to create work item', err);
        alert('Failed to create work item – see console for details.');
        return;
      }

      const wi = await resp.json();
      onTaskCreated?.(wi);

      // Navigate to the new Work Item detail page if you have a route for it:
      // e.g. `/work-items/${wi.id}`
      window.location.href = `/work-items/${wi.id}`;
    } finally {
      setCreating(false);
    }
  }

  return (
    <div className="flex flex-col items-center justify-center py-12 text-center">
      <p className="mb-4 text-sm text-muted-foreground">No tasks created yet.</p>
      <Button onClick={handleCreateWorkItem} disabled={creating}>
        {creating ? 'Creating…' : '＋ New Work Item'}
      </Button>
    </div>
  );
}
```

Then in your main `ProjectTasks` component, where you currently render the “No tasks” empty state, swap it for:

```tsx
{tasks.length === 0 ? (
  <ProjectTasksEmptyState
    projectId={projectId}
    onTaskCreated={(wi) => {
      // optional: update local state instead of full reload
      setTasks((prev) => [...prev, wi]);
    }}
  />
) : (
  /* existing list/board rendering */
)}
```

> After this patch: on the **Tasks (0)** tab you’ll see a big
> “＋ New Work Item” button instead of just “No tasks created yet.”
