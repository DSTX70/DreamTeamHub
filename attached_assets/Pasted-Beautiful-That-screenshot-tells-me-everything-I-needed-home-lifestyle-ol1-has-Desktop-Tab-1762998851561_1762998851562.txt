```ts
// app/server/fcc/sku-map.server.ts
import { db } from "~/server/db"; // adjust import to match your project
import { sku_mappings } from "~/server/db/schema"; // adjust path/name if different

export type FccSkuRow = {
  id: string | number;
  brand: string;
  shot_key: string;
  base_key: string | null;
  label: string | null;
};

export type FccSkuMap = Record<string, FccSkuRow>;

export async function getFccSkuMap(): Promise<FccSkuMap> {
  // Pull only Fab Card Co rows
  const rows = (await db
    .select()
    .from(sku_mappings)
    .where(sku_mappings.brand.eq("fcc"))) as FccSkuRow[];

  const map: FccSkuMap = {};

  for (const row of rows) {
    if (!row.shot_key) continue;
    map[row.shot_key] = row;
  }

  return map;
}
```
